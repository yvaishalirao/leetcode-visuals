Top K Frequent Words

View on LeetCode ↗
Given an array of strings words and an integer k, return the k most frequent strings, sorted by frequency from highest to lowest. Words with the same frequency are sorted in lexicographical order.
Solution code
class Solution(object):
    def topKFrequent(self, words, k):
        dict1 = {}

        for i in words:
            dict1[i] = dict1.get(i, 0) + 1

        sorted_items = sorted(dict1.items(), key=lambda item: (-item[1], item[0]))

        ans = []
        for i in sorted_items:
            ans.append(i[0])
            k -= 1
            if k == 0:
                break

        return ans
