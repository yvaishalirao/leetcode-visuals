Sequential Digits

View on LeetCode ↗
An integer has sequential digits if each digit is one more than the previous digit. Return a sorted list of all integers in [low, high] that have sequential digits.
Solution code
class Solution(object):
    def sequentialDigits(self, low, high):
        ans = []

        s = "123456789"
        l = str(low)
        h = str(high)

        for length in range(len(l), len(h) + 1):
            for start in range(0, 10 - length):
                num = int(s[start:start + length])
                if low <= num <= high:
                    ans.append(num)

        return ans
